Neste exemplo vou demonstrar um treho de codigo fonte 
import pandas as pd
        import numpy as np
        
        def remove_outlier(df_old):
            # Copia o DataFrame original para evitar modificar o original diretamente
            df = df_old.copy()
            
            # Calcula o Z-Score de cada valor nas colunas numéricas
            from scipy.stats import zscore
            
            # Aplica o zscore apenas nas colunas numéricas
            num_cols = df.select_dtypes(include=np.number).columns
            
            # Calcula o Z-Score para cada coluna numérica
            z_scores = np.abs(zscore(df[num_cols]))
            
            # Cria uma máscara para filtrar os outliers (valores com Z-Score maior que 3)
            mask = (z_scores < 3).all(axis=1)
            
            # Aplica a máscara no DataFrame
            df_cleaned = df[mask]
            
            return df_cleaned
